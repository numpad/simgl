GLuint VBO, VBO2, VAO, EBO;
glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO);
glGenBuffers(1, &EBO);
glGenBuffers(1, &VBO2);

glBindVertexArray(VAO);
glBindBuffer(GL_ARRAY_BUFFER, VBO);
glBufferData(GL_ARRAY_BUFFER, obj_vertices.size() * sizeof(glm::vec3), &(obj_vertices[0]), GL_STATIC_DRAW);
glBindBuffer(GL_ARRAY_BUFFER, VBO2);
glBufferData(GL_ARRAY_BUFFER, obj_normals.size() * sizeof(glm::vec3), &(obj_normals[0]), GL_STATIC_DRAW);
glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, obj_indices.size() * sizeof(GLuint), &(obj_indices[0]), GL_STATIC_DRAW);

glBindBuffer(GL_ARRAY_BUFFER, VBO);
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(GLfloat), (void *)0);
glEnableVertexAttribArray(0);

glBindBuffer(GL_ARRAY_BUFFER, VBO2);
glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(GLfloat), (void *)0);
glEnableVertexAttribArray(1);



glUseProgram(shader_program);
		
glActiveTexture(GL_TEXTURE0);
glBindTexture(GL_TEXTURE_2D, tex_thing);

glUniformMatrix4fv(uModel, 1, GL_FALSE, glm::value_ptr(model));
glUniformMatrix4fv(uMVP, 1, GL_FALSE, glm::value_ptr(MVP));

glBindVertexArray(VAO);

glDrawElements(GL_TRIANGLES, obj_indices.size(), GL_UNSIGNED_INT, 0);
//glDrawArrays(GL_TRIANGLES, 0, vertices.size());
